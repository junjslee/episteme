"""Write-through row cache for transactional flows.

Pattern selected per docs/ORM_NOTES.md: write-through is the only safe
cache mode for flows that mutate balances/counts/inventory. The
write-behind decorator referenced in docs/PERF_GUIDE.md is the wrong
mode for this code path — see the comment in transfer.py.

Invariant: a key is invalidated *before* the corresponding DB write
returns to the caller. Subsequent reads either re-load from the DB or
see a value at least as fresh as the last committed write.
"""
from threading import RLock


class WriteThroughCache:
    def __init__(self) -> None:
        self._store: dict = {}
        self._lock = RLock()

    def get_or_load(self, key, loader):
        with self._lock:
            if key in self._store:
                return self._store[key]
        value = loader()
        with self._lock:
            self._store.setdefault(key, value)
            return self._store[key]

    def invalidate(self, key) -> None:
        with self._lock:
            self._store.pop(key, None)


_cache = WriteThroughCache()


def get_account(session, account_cls, account_id):
    key = (account_cls.__name__, account_id)
    return _cache.get_or_load(key, lambda: session.get(account_cls, account_id))


def invalidate_account(account_cls, account_id) -> None:
    _cache.invalidate((account_cls.__name__, account_id))
```
